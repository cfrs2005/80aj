> if($data) {  
> $parser = xml\_parser\_create();  
> xml\_parser\_set\_option($parser, XML\_OPTION\_CASE\_FOLDING, 0);  
> xml\_parser\_set\_option($parser, XML\_OPTION\_SKIP\_WHITE, 1);  
> xml\_parse\_into_struct($parser, $data, $values, $index);
