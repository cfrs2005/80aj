**discuzUbb代码过滤**

<pre class="brush: php; title: ; notranslate" title="">function _ubb_fillter($string){
	$string = preg_replace(
			array('/&amp;(#\d{3,5};)/', "/\[hide=?\d*\](.*?)\[\/hide\]/is", "/\[\/?\w+=?.*?\]/"),
			array('&\\1','&lt;b&gt;**** Hidden Message *****&lt;/b&gt;',''),
			$string);
	return $string;
}
</pre>
